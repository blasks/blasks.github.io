\ifdefined\titleformat
  \titleformat{\section}

        \noindent\makebox[\linewidth][c]{\ensuremath{\diamond\ \diamond\ \diamond}}\\[2pt]%

    }

  \titleformat{\subsection}
    {\headingfont\normalsize\color{textcolor}}

    {0pt}

    }%

    }%
  }
\fi
